package com.tanuj;

import java.util.Scanner;

public class Main {
	public static void main(String[] args) {

		Scanner sc = new Scanner(System.in);
		String inputStr = sc.next();

		if (inputStr.length() != 18) {
			System.out.println("Invalid Input");
			return;
		}

		// checking dept Code
		StringBuilder sb = new StringBuilder(inputStr);
		String deptCode = sb.substring(0, 3).toString();
		int deptC = Integer.parseInt(deptCode);
		if (deptC != 101 && deptC != 102 && deptC != 103) {
			System.out.println("Invalid Department Code");
			return;
		} else {
			System.out.println("Department Code: " + deptCode);
		}

		// checking publication year
		String pubYear = sb.substring(3, 7);
		int publications = Integer.parseInt(pubYear);
		if (publications >= 1900 && publications <= 2020) {
			System.out.println("Year of Publication: " + publications);
		} else {
			System.out.println("Invalid Year");
			return;
		}

		// checking the number of pages
		String noOfpages = sb.substring(7, 12);
		int countPage = Integer.valueOf(noOfpages);
		if (countPage < 10) {
			System.out.println("Invalid Pages");
			return;
		} else {
			System.out.println("Number of Pages: " + countPage);
		}

		// checking book id
		String bookId = sb.substring(12, 18);
		if (bookId.matches("^[A-Za-z][0-9]+$")) {
			System.out.println("Book Id: " + bookId);
		} else {
			System.out.println("Invalid Book Id");
		}

	}
}
